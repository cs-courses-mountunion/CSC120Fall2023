 12 import java.awt.*;
 13 import javax.swing.*;
 14 
 15 public class MUPanel extends JPanel {
 16 
 17     // Declare private data here
 18     
 19     private LollipopTree aTree, tree2, lastTree, iLiedThisIsTheRealLastOne;
 20     private SmileySun theSun, tatooineSun, sol;
 21     
 22     public MUPanel() {
 23         setLayout(null);
 24         setPreferredSize(new Dimension(900, 400));
 25         setName("CSC 120 Lecture 7 -- Differently-sized Suns");
 26         setBackground(Color.WHITE); 
 27 
 28         // Instantiate objects and initialize data here
 29         
 30         aTree = new LollipopTree(480, 80);
 31         tree2 = new LollipopTree(650, 120);
 32         lastTree = new LollipopTree(750, 40);
 33         iLiedThisIsTheRealLastOne = new LollipopTree(420, 150);
 34         
 35         theSun = new SmileySun(60, 60, Color.BLUE, 90);
 36         tatooineSun = new SmileySun(150, 20, new Color(0, 200, 0), 50);
 37         sol = new SmileySun(30, 160, Color.RED, 130);
 38         
 39     } // end of MUPanel constructor
 40     
 41 
 42     public void paintComponent(Graphics g) {
 43         super.paintComponent(g);
 44         
 45         // The house
 46         g.setColor(Color.BLACK);
 47         g.drawRect(200, 150, 200, 100);
 48                 
 49         // The roof
 50         g.drawLine(200, 150, 300, 90);
 51         g.drawLine(300, 90, 400, 150);
 52 
 53         // The window
 54         g.setColor(Color.YELLOW);
 55         g.fillRect(225, 180, 90, 40);
 56         g.setColor(Color.BLACK);
 57         g.drawRect(225, 180, 90, 40);
 58         g.drawLine(225, 200, 315, 200);
 59         g.drawLine(270, 180, 270, 220);
 60                 
 61                 
 62         // The door and doorknob
 63         g.setColor(Color.BLUE);
 64         g.fillRect(350, 190, 30, 60);
 65         g.setColor(Color.ORANGE);
 66         g.fillOval(365, 220, 10, 10);
 67 
 68 
 69         // The tree
 70         aTree.draw(g);
 71         tree2.draw(g);
 72         lastTree.draw(g);
 73         iLiedThisIsTheRealLastOne.draw(g);
 74     
 75         // The sun
 76         theSun.draw(g);
 77         tatooineSun.draw(g);
 78         sol.draw(g);
 79 
 80     } // end of paintComponent()
 81 
 82     
 83     
 84     /***********************************************
 85      * Do NOT change or delete anything below here!
 86      ***********************************************/
 87     public void frame()
 88     {
 89         for (Component c: getComponents())
 90             c.setSize(c.getPreferredSize());
 91         JFrame f = new JFrame(getName());
 92         f.setContentPane(this);
 93         f.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
 94         f.pack();
 95         f.setVisible(true);   
 96         
 97     } // end of frame()
 98 
 99     public static void main(String args[]){new MUPanel().frame();}
100 
101 }
