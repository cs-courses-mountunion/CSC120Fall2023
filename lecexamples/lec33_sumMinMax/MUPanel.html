126     public void processSoccerStats(){
127         
128         SoccerTeam[] team;
129         
130         team = new SoccerTeam[10];
131         team[0] = new SoccerTeam(2023, 5, 10);
132         team[1] = new SoccerTeam(2022, 7, 6);
133         team[2] = new SoccerTeam(2021, 10, 7);
134         team[3] = new SoccerTeam(2019, 9, 10);
135         team[4] = new SoccerTeam(2018, 12, 7);
136         team[5] = new SoccerTeam(2017, 12, 6);
137         team[6] = new SoccerTeam(2016, 12, 5);
138         team[7] = new SoccerTeam(2015, 9, 9);
139         team[8] = new SoccerTeam(2014, 4, 14);
140         team[9] = new SoccerTeam(2013, 6, 13);
141 
142         outputArea.append("----------------------------\n\n");
143         for (Integer t = 0; t < team.length; t++) {
144             outputArea.append(team[t].toString() + "\n");
145         }
146         outputArea.append("\n\n");
147         
148         
149         
150         
151         Integer totalWins = 0;
152         Integer maxLosses = Integer.MIN_VALUE;
153         Integer maxPosition = 0;
154         Integer minLosses = Integer.MAX_VALUE;
155         Integer minPosition = 0;
156         
157         for (Integer t = 0; t < team.length; t++) {
158             totalWins = totalWins + team[t].getWins();
159             
160             if (team[t].getLosses() < minLosses) {
161                 minLosses = team[t].getLosses();
162                 minPosition = t;
163             }
164             
165             if (team[t].getLosses() > maxLosses) {
166                 maxLosses = team[t].getLosses();
167                 maxPosition = t;
168             }
169         } // end for
170         
171         outputArea.append("Total wins over the years = " + totalWins + "\n");
172         
173         Double avgWins = 1.0 * totalWins / team.length;
174         outputArea.append("Average wins per year = " + avgWins + "\n");
175         outputArea.append("\n");
176         outputArea.append("Most losses in any year = " + maxLosses + "\n");
177           outputArea.append("\tThat was the team from year " + team[maxPosition].getYear() + "\n");
178         outputArea.append("Fewest losses in any year = " + minLosses + "\n");
179           outputArea.append("\tThat was the team from year " + team[minPosition].getYear() + "\n");
180         
181     }
